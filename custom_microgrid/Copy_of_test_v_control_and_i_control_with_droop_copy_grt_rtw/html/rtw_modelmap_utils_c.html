61/** Function: rtwCAPI_EncodePath ===============================================



62 *  Abstract:



63 *     Escape all '|' characters in bpath. For examples 'aaa|b' will become



64 *     'aaa~|b'. The caller is responsible for freeing the returned string



65 *



66 *



67 * NOTE: returned string can be NULL in two cases:



68 *     (1) string passed in was NULL



69 *     (2) a memory allocation error occurred



70 * In the second case, the caller need to report the error



71 */



72char* rtwCAPI_EncodePath(const char* path)



73{



74    char* encodedPath     = NULL;



75    size_t pathLen        = (path==NULL) ? 0:strlen(path) + 1;



76    size_t encodedPathLen = pathLen;



77    unsigned i;



78    unsigned j = 0;



79



80    if (path == NULL) return NULL;



81



82    for (i = 0; i < pathLen; ++i) {



83        if (path[i] == '|' || path[i] == '~') ++encodedPathLen;



84    }



85



86    encodedPath = (char_T*)utMalloc(encodedPathLen*sizeof(char_T));



87    if (encodedPath == NULL) return encodedPath;



88



89    for (i = 0; i < pathLen; ++i) {



90        char ch = path[i];



91        if (ch == '~' || ch == '|') encodedPath[j++] = '~';



92        encodedPath[j++] = ch;



93    }



94    utAssert(j == encodedPathLen);



95    utAssert(encodedPath[j-1] == '\0');



96



97    return encodedPath;



98



99} /* rtwCAPI_EncodePath */

101/** Function: rtwCAPI_GetSigAddrFromMap ========================================



102 *



103 */



104void rtwCAPI_GetSigAddrFromMap(uint8_T      isPointer,



105                               int_T*       sigComplexity,



106                               int_T*       sigDataType,



107                               void**       sigDataAddr,



108                               int_T*       sigIdx,



109                               uint_T       mapIdx,



110                               void**       dataAddrMap)



111{



112    if (isPointer) {



113        /* Dereference pointer and cache the address */



114



115        /* Imported Pointers cannot be complex - Assert */



116        utAssert(sigComplexity[*sigIdx] != 1);



117        UNUSED_PARAMETER(sigComplexity);



118



119        /* Check for data type and dereference accordingly */



120        switch (sigDataType[*sigIdx]) {



121          case SS_DOUBLE:



122            sigDataAddr[*sigIdx] = \



123                (void*) *((real_T **) dataAddrMap[mapIdx]);



124            break;



125          case SS_SINGLE:



126            sigDataAddr[*sigIdx] = \



127                (void*) *((real32_T **) dataAddrMap[mapIdx]);



128            break;



129          case SS_UINT32:



130            sigDataAddr[*sigIdx] = \



131                (void*) *((uint32_T **) dataAddrMap[mapIdx]);



132            break;



133          case SS_INT32:



134            sigDataAddr[*sigIdx] = \



135                (void*) *((int32_T **) dataAddrMap[mapIdx]);



136            break;



137          case SS_UINT16:



138            sigDataAddr[*sigIdx] = \



139                (void*) *((uint16_T **) dataAddrMap[mapIdx]);



140            break;



141          case SS_INT16:



142            sigDataAddr[*sigIdx] = \



143                (void*) *((int16_T **) dataAddrMap[mapIdx]);



144            break;



145          case SS_UINT8:



146            sigDataAddr[*sigIdx] = \



147                (void*) *((uint8_T **) dataAddrMap[mapIdx]);



148            break;



149          case SS_INT8:



150            sigDataAddr[*sigIdx] = \



151                (void*) *((int8_T **) dataAddrMap[mapIdx]);



152            break;



153          case SS_BOOLEAN:



154            sigDataAddr[*sigIdx] = \



155                (void*) *((boolean_T **) dataAddrMap[mapIdx]);



156            break;



157          default:



158            sigDataAddr[*sigIdx] = \



159                (void*) *((real_T **) dataAddrMap[mapIdx]);



160            break;



161        }  /* end switch */



162    } else {



163        /* if Data is not a pointer store the address directly */



164        sigDataAddr[*sigIdx] = dataAddrMap[mapIdx];



165    }



166



167} /* rtwCAPI_GetSigAddrFromMap */

170/** Function: rtwCAPI_HasStates ================================================



171 *



172 */



173boolean_T rtwCAPI_HasStates(const rtwCAPI_ModelMappingInfo* mmi)



174{



175    int_T i;



176    int_T nCMMI;



177



178    if (mmi == NULL) return(0U);



179



180    if (rtwCAPI_GetNumStates(mmi) > 0) return(1U);



181



182    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



183    for (i = 0; i < nCMMI; ++i) {



184        if (rtwCAPI_HasStates(rtwCAPI_GetChildMMI(mmi,i))) return(1U);



185    }



186    return(0U);



187



188} /* rtwCAPI_HasStates */

250/** Function: rtwCAPI_GetNumContStateRecords ===================================



251 *



252 */



253int_T rtwCAPI_GetNumContStateRecords(const rtwCAPI_ModelMappingInfo* mmi)



254{



255    int_T i;



256    int_T nRecs;



257    int_T nCMMI;



258    int_T nCStateRecs;



259    const rtwCAPI_States* states;



260    const rtwCAPI_DataTypeMap* dataTypeMap;



261



262    if (mmi == NULL) return 0;



263



264    nCStateRecs = 0;



265    states      = rtwCAPI_GetStates(mmi);



266    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



267



268    nRecs = rtwCAPI_GetNumStates(mmi);



269    for (i = 0; i < nRecs; i++) {



270        if (rtwCAPI_IsAContinuousState(states,i)) {



271            ++nCStateRecs;



272



273            /* All continuous states should be able to be logged to MAT-File



274             * so we do not need to skip any states here. */



275            utAssert(rtwCAPI_CanLogStateToMATFile(dataTypeMap, states, i));



276        }



277    }



278



279    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



280    for (i = 0; i < nCMMI; ++i) {



281        const rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



282        nCStateRecs += rtwCAPI_GetNumContStateRecords(cMMI);



283    }



284    return nCStateRecs;



285



286} /* rtwCAPI_GetNumContStateRecords */

314/** Function: rtwCAPI_UpdateFullPaths =========================================*



315 *



316 */



317const char_T* rtwCAPI_UpdateFullPaths(rtwCAPI_ModelMappingInfo* mmi,



318                                      const char_T* path,



319                                      boolean_T isCalledFromTopModel)



320{



321    int_T         i;



322    int_T         nCMMI;



323    size_t        pathLen;



324    char_T*       mmiPath;



325    size_t        mmiPathLen;



326    char_T*       relMMIPath;



327    size_t        relMMIPathLen;



328



329    if (mmi == NULL) return NULL;



330



331    utAssert(path != NULL);



332    utAssert( rtwCAPI_GetFullPath(mmi) == NULL );



333



334    pathLen = strlen(path)+1;



335



336    if (isCalledFromTopModel) {



337        /* If called from top model - FullPath is same as path */



338        mmiPath = (char_T*)utMalloc(pathLen*sizeof(char_T));



339        (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));



340    }



341    else {        



342        relMMIPath = rtwCAPI_EncodePath(rtwCAPI_GetPath(mmi));



343        if ( (relMMIPath== NULL) && (rtwCAPI_GetPath(mmi) != NULL)) {



344            return rtwCAPI_mallocError;



345        }



346        relMMIPathLen = relMMIPath ? (strlen(relMMIPath) + 1) : 0;



347        



348        mmiPathLen = pathLen + relMMIPathLen;



349        



350        mmiPath = (char_T*)utMalloc(mmiPathLen*sizeof(char_T));



351        if (mmiPath == NULL) return rtwCAPI_mallocError;



352        (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));



353        utAssert(mmiPath[pathLen-1] == '\0');



354        



355        if (relMMIPath) {



356            /* mmiPath = path + | + relMMIPath + '\0' */



357            mmiPath[pathLen-1] = '|';



358            (void)memcpy(&(mmiPath[pathLen]),



359                         relMMIPath, relMMIPathLen*sizeof(char_T));



360            utAssert(mmiPath[mmiPathLen-1] == '\0');



361            utFree(relMMIPath);



362        }



363    }



364    rtwCAPI_SetFullPath(*mmi, mmiPath);



365



366    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



367    for (i = 0; i < nCMMI; ++i) {



368        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



369        const char_T* errstr = rtwCAPI_UpdateFullPaths(cMMI, mmiPath, 0);



370        if (errstr != NULL) return errstr;



371    }



372    return NULL;



373



374} /* rtwCAPI_UpdateFullPaths */

377/** Function: rtwCAPI_GetFullStateBlockPath ===================================



378 *



379 */



380char* rtwCAPI_GetFullStateBlockPath(const char* stateBlockPath,



381                                           const char* mmiPath,



382                                           size_t      mmiPathLen,



383                                           boolean_T   crossingModel)



384{



385    char_T* blockPath          = NULL;



386    char_T* fullStateBlockPath = NULL;



387    size_t     fullStateBlockPathLen;



388



389    if (stateBlockPath == NULL) goto EXIT_POINT;



390



391    /* fullStateBlockPath = mmiPath + | + blockPath + '\0' */



392    /* If crossing a model boundary encode, otherwise do not */



393



394    if (crossingModel) {



395        blockPath = rtwCAPI_EncodePath(stateBlockPath);



396        if (blockPath == NULL) goto EXIT_POINT;



397    } else {



398        size_t len = strlen(stateBlockPath)+1;



399        blockPath = (char*)utMalloc(len*sizeof(char));



400        if (blockPath == NULL) goto EXIT_POINT;



401        (void)strcpy(blockPath,stateBlockPath);



402    }



403    utAssert(blockPath != NULL);



404    fullStateBlockPathLen = ( (mmiPath==NULL) ?



405                              strlen(blockPath) + 1 :



406                              mmiPathLen + strlen(blockPath) + 2 );



407    fullStateBlockPath    = (char*)utMalloc(fullStateBlockPathLen*sizeof(char));



408    if (fullStateBlockPath == NULL) goto EXIT_POINT;



409



410    if (mmiPath != NULL) {



411        (void)strcpy(fullStateBlockPath, mmiPath);



412        fullStateBlockPath[mmiPathLen]   = '|';



413        fullStateBlockPath[mmiPathLen+1] =  '\0';



414        (void)strcat(fullStateBlockPath, blockPath);



415    } else {



416        (void)strcpy(fullStateBlockPath, blockPath);



417        fullStateBlockPath[fullStateBlockPathLen-1] =  '\0';



418    }



419    utAssert(fullStateBlockPath[fullStateBlockPathLen-1] == '\0');



420



421  EXIT_POINT:



422    utFree(blockPath);



423    return fullStateBlockPath; /* caller is responsible for free */



424}

426uint_T rtwCAPI_GetStateWidth(const rtwCAPI_DimensionMap* dimMap,



427                             const uint_T*               dimArray,



428                             const rtwCAPI_States*       states,



429                             uint_T                      iState)



430{



431    uint_T mapIdx = rtwCAPI_GetStateDimensionIdx(states, iState);



432    utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );



433    mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);



434    return (dimArray[mapIdx] * dimArray[mapIdx+1]);



435}

439/** Function: rtwCAPI_GetStateRecordInfo =======================================



440 *



441 */



442const char_T* rtwCAPI_GetStateRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,



443                                         const char_T**    sigBlockName,



444                                         const char_T**    sigLabel,



445                                         const char_T**    sigName,



446                                         int_T*            sigWidth,



447                                         int_T*            sigDataType,



448                                         int_T*            logDataType,



449                                         int_T*            sigComplexity,



450                                         void**            sigDataAddr,



451                                         boolean_T*        sigCrossMdlRef,



452                                         boolean_T*        sigInProtectedMdl,



453                                         const char_T**    sigPathAlias,



454                                         real_T*           sigSampleTime,



455                                         int_T*            sigHierInfoIdx,



456                                         uint_T*           sigFlatElemIdx,                                         



457                                         const rtwCAPI_ModelMappingInfo** sigMMI,



458                                         int_T*            sigIdx,



459                                         boolean_T         crossingModel,



460                                         boolean_T         isInProtectedMdl,



461                                         real_T*           contStateDeriv,



462                                         boolean_T         rtwLogging)



463{



464    int_T               i;



465    int_T               nCMMI;



466    int_T               nStates;



467    const char_T*       mmiPath;



468    size_t              mmiPathLen;



469    const rtwCAPI_States*  states;



470    const rtwCAPI_DimensionMap* dimMap;



471    const uint_T*       dimArray;



472    const rtwCAPI_DataTypeMap*  dataTypeMap;



473    void**              dataAddrMap;



474    const char_T*       errstr = NULL;



475    uint8_T             isPointer = 0;



476



477    if (mmi == NULL) goto EXIT_POINT;



478    isInProtectedMdl = isInProtectedMdl || rtwCAPI_IsProtectedModel(mmi);



479    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



480    for (i = 0; i < nCMMI; ++i) {



481        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



482        real_T* childContStateDeriv = NULL;



483



484        if (cMMI == NULL) continue;



485



486        if (contStateDeriv) {



487            int idx;



488



489            idx = rtwCAPI_MMIGetContStateStartIndex(cMMI);



490            if(idx == -1) continue;



491            



492            childContStateDeriv = &contStateDeriv[idx];



493        }



494        errstr = rtwCAPI_GetStateRecordInfo(cMMI,



495                                            sigBlockName,



496                                            sigLabel,



497                                            sigName,



498                                            sigWidth,



499                                            sigDataType,



500                                            logDataType,



501                                            sigComplexity,



502                                            sigDataAddr,



503                                            sigCrossMdlRef,



504                                            sigInProtectedMdl,



505                                            sigPathAlias,



506                                            sigSampleTime,



507                                            sigHierInfoIdx,



508                                            sigFlatElemIdx,



509                                            sigMMI,



510                                            sigIdx,



511                                            0x1, /* true, */



512                                            isInProtectedMdl,



513                                            childContStateDeriv,



514                                            rtwLogging);



515        if (errstr != NULL) goto EXIT_POINT;



516    }



517



518    nStates = rtwCAPI_GetNumStates(mmi);



519    if (nStates < 1) goto EXIT_POINT;



520



521    mmiPath     = rtwCAPI_GetFullPath(mmi);



522    mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);



523    states      = rtwCAPI_GetStates(mmi);



524    dimMap      = rtwCAPI_GetDimensionMap(mmi);



525    dimArray    = rtwCAPI_GetDimensionArray(mmi);



526    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



527    dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);



528



529    for (i = 0; i < nStates; ++i) {



530        uint_T      mapIdx;



531



532        /* If collecting continuous states, skip non-continuous states */



533        if (contStateDeriv && !rtwCAPI_IsAContinuousState(states,i)) continue;



534



535        /* For RTW logging, skip states that cannot be logged to MAT-File. */



536        if ((rtwLogging) &&



537            (rtwCAPI_CanLogStateToMATFile(dataTypeMap, states, i) == false)) continue;



538



539        /* BlockPath (caller is responsible for free) */



540        sigBlockName[*sigIdx] =



541            rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStateBlockPath(states,i),



542                                          mmiPath, mmiPathLen, crossingModel);



543        if (sigBlockName[*sigIdx] == NULL) {



544            errstr = rtwCAPI_mallocError;



545            goto EXIT_POINT;



546        }



547



548        /* Label */



549        if (rtwCAPI_IsAContinuousState(states,i)){



550            sigLabel[*sigIdx] = "CSTATE";



551        } else {



552            sigLabel[*sigIdx] = "DSTATE";



553        }



554        sigName[*sigIdx] = rtwCAPI_GetStateName(states, i);



555



556        /* Width */



557        sigWidth[*sigIdx] = rtwCAPI_GetStateWidth(dimMap, dimArray, states, i);



558



559        /* DataType and logDataType */



560        mapIdx = rtwCAPI_GetStateDataTypeIdx(states, i);



561        sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);



562        /* this mimics code in simulink.dll:DtGetDataTypeLoggingId() */



563        if (logDataType) {



564            switch (sigDataType[*sigIdx]) {



565              case SS_DOUBLE:



566              case SS_SINGLE:



567              case SS_INT8:



568              case SS_UINT8:



569              case SS_INT16:



570              case SS_UINT16:



571              case SS_INT32:



572              case SS_UINT32:



573              case SS_BOOLEAN:



574                logDataType[*sigIdx] = sigDataType[*sigIdx];



575                break;



576              case SS_ENUM_TYPE:



577                logDataType[*sigIdx] = SS_INT32;



578                break;



579              default:



580                logDataType[*sigIdx] = SS_DOUBLE;



581                break;



582            }



583        }



584



585        /* Complexity */



586        sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);



587



588        /* Data Access - Pointer or Direct*/



589        isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));



590



591        /* Address */



592        if (contStateDeriv) {



593            int_T csvIdx = rtwCAPI_GetContStateStartIndex(states,i);



594            utAssert(csvIdx >= 0);



595            sigDataAddr[*sigIdx] = &contStateDeriv[csvIdx];



596        } else {



597            mapIdx = rtwCAPI_GetStateAddrIdx(states,i);



598            rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,



599                                      sigDataAddr, sigIdx, mapIdx, dataAddrMap);



600        }



601



602        /* CrossingModelBoundary */



603        sigCrossMdlRef[*sigIdx] = crossingModel;



604



605        if (sigInProtectedMdl)



606        {



607            sigInProtectedMdl[*sigIdx] = isInProtectedMdl;



608        }



609



610        if (sigPathAlias && 



611            rtwCAPI_GetStatePathAlias(states,i) != NULL && 



612            rtwCAPI_GetStatePathAlias(states,i)[0] != '\0') {



613            sigPathAlias[*sigIdx] =  



614                rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStatePathAlias(states,i),



615                                              mmiPath, mmiPathLen, crossingModel);



616        }



617        



618        /* Sample Time */



619        if (sigSampleTime) {



620            const rtwCAPI_SampleTimeMap* tsMap = rtwCAPI_GetSampleTimeMap(mmi);



621            int_T TID;



622            mapIdx = rtwCAPI_GetStateSampleTimeIdx(states, i);



623            TID = rtwCAPI_GetSampleTimeTID(tsMap, mapIdx);



624            if (TID >= 0) {



625                sigSampleTime[2*(*sigIdx)] =



626                    *((const real_T*)rtwCAPI_GetSamplePeriodPtr(tsMap,mapIdx));



627                sigSampleTime[2*(*sigIdx)+1] =



628                    *((const real_T*)rtwCAPI_GetSampleOffsetPtr(tsMap,mapIdx));



629            } else { /* triggered */



630                utAssert(TID==-1); 



631                sigSampleTime[2*(*sigIdx)]   = -1.0;



632                sigSampleTime[2*(*sigIdx)+1] = -1.0;



633            }



634        }



635



636        /* HierInfoIdx and FlatElemIdx */



637        if (sigHierInfoIdx && sigFlatElemIdx)



638        {



639            sigHierInfoIdx[*sigIdx] = rtwCAPI_GetStateHierInfoIdx(states, i);



640            sigFlatElemIdx[*sigIdx] = rtwCAPI_GetStateFlatElemIdx(states, i);



641        }      



642



643        /* MMI for each state */



644        if (sigMMI)



645        {



646            sigMMI[*sigIdx] = mmi;



647        }        



648



649        ++(*sigIdx);



650    }



651



652  EXIT_POINT:



653    return(errstr);



654



655} /* rtwCAPI_GetStateRecordInfo */

659/** Function: rtwCAPI_GetNumSigLogRecords ======================================



660 *



661 */



662int_T rtwCAPI_GetNumSigLogRecords(const rtwCAPI_ModelMappingInfo* mmi)



663{



664    int_T i;



665    int_T nRecs;



666    int_T nCMMI;



667



668    if (mmi == NULL) return(0);



669



670    nRecs = rtwCAPI_GetNumSignals(mmi);



671



672    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



673    for (i = 0; i < nCMMI; ++i) {



674        const rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



675        nRecs += rtwCAPI_GetNumSigLogRecords(cMMI);



676    }



677    return(nRecs);



678



679} /* rtwCAPI_GetNumSigLogRecords */

685int_T rtwCAPI_GetNumSigLogRecordsForRTWLogging(const rtwCAPI_ModelMappingInfo* mmi)



686{



687    int_T i;



688    int_T nRecs = 0;



689    int_T nSignals = 0;



690    int_T nCMMI;



691    const rtwCAPI_Signals *signals = NULL;



692    const rtwCAPI_DataTypeMap* dataTypeMap;



693



694    if (mmi == NULL) return(0);



695



696    nSignals = rtwCAPI_GetNumSignals(mmi);



697    signals = rtwCAPI_GetSignals(mmi);



698    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



699



700    for (i = 0; i < nSignals; ++i) {

703        }



704    }



705



706    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);

708        const rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



709        nRecs += rtwCAPI_GetNumSigLogRecordsForRTWLogging(cMMI);



710    }



711



712    return(nRecs);

717/** Function: rtwCAPI_GetSigLogRecordInfo ======================================



718 *



719 */



720const char_T* rtwCAPI_GetSigLogRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,



721                                          const char_T**    sigBlockName,



722                                          const char_T**    sigLabel,



723                                          int_T*            sigWidth,



724                                          int_T*            sigDataType,



725                                          int_T*            logDataType,



726                                          int_T*            sigComplexity,



727                                          void**            sigDataAddr,



728                                          boolean_T*        sigCrossMdlRef,



729                                          int_T*            sigIdx,



730                                          boolean_T         crossingModel,



731                                          boolean_T         rtwLogging)



732{



733    int_T               i;



734    int_T               nCMMI;



735    int_T               nSignals;



736    const char_T*       mmiPath;



737    size_t              mmiPathLen;



738    const rtwCAPI_Signals*  signals;



739    const rtwCAPI_DimensionMap* dimMap;



740    const uint_T*       dimArray;



741    const rtwCAPI_DataTypeMap*  dataTypeMap;



742    void**              dataAddrMap;



743    const char_T*       errstr = NULL;



744    uint8_T             isPointer = 0;



745    char*               blockPath = NULL;



746



747    if (mmi == NULL) goto EXIT_POINT;



748



749    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



750    for (i = 0; i < nCMMI; ++i) {



751        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



752



753        errstr = rtwCAPI_GetSigLogRecordInfo(cMMI,



754                                             sigBlockName,



755                                             sigLabel,



756                                             sigWidth,



757                                             sigDataType,



758                                             logDataType,



759                                             sigComplexity,



760                                             sigDataAddr,



761                                             sigCrossMdlRef,



762                                             sigIdx,



763                                             true,



764                                             rtwLogging);



765        if (errstr != NULL) goto EXIT_POINT;



766    }



767



768    nSignals = rtwCAPI_GetNumSignals(mmi);



769    if (nSignals < 1) goto EXIT_POINT;



770



771    mmiPath     = rtwCAPI_GetFullPath(mmi);



772    mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);



773    signals     = rtwCAPI_GetSignals(mmi);



774    dimMap      = rtwCAPI_GetDimensionMap(mmi);



775    dimArray    = rtwCAPI_GetDimensionArray(mmi);



776    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



777    dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);



778



779    for (i = 0; i < nSignals; ++i) {



780        uint_T mapIdx;



781        size_t sigPathLen;



782        char*  sigPath;



783



784        /* For RTW logging, skip states that cannot be logged to MAT-File. */



785        if ((rtwLogging) &&



786            (rtwCAPI_CanLogSignalToMATFile(dataTypeMap, signals, i) == false)) continue;



787



788        /* sigBlockPath = mmiPath + | + BlockPath + '\0' */



789        /* If crossing a model boundary encode, otherwise do not */



790



791        if (crossingModel) {



792            blockPath = rtwCAPI_EncodePath(rtwCAPI_GetSignalBlockPath(signals, i));



793            if ( (blockPath == NULL) &&



794                 (rtwCAPI_GetSignalBlockPath(signals, i) != NULL)) {



795                errstr = rtwCAPI_mallocError;



796                goto EXIT_POINT;



797            }



798        } else {



799            const char* constBlockPath = rtwCAPI_GetSignalBlockPath(signals, i);



800            blockPath = (char*)utMalloc((strlen(constBlockPath)+1)*sizeof(char));



801            (void)strcpy(blockPath, constBlockPath);



802        }



803        utAssert(blockPath != NULL);



804        sigPathLen = ( (mmiPath==NULL) ?



805                                   strlen(blockPath) + 1 :



806                                   mmiPathLen + strlen(blockPath) + 2 );



807        sigPath    = (char*)utMalloc(sigPathLen*sizeof(char));



808        if (sigPath == NULL) {



809            errstr = rtwCAPI_mallocError;



810            goto EXIT_POINT;



811        }



812        if (mmiPath != NULL) {



813            (void)strcpy(sigPath, mmiPath);



814            sigPath[mmiPathLen]   = '|';



815            sigPath[mmiPathLen+1] =  '\0';



816            (void)strcat(sigPath, blockPath);



817        } else {



818            (void)strcpy(sigPath, blockPath);



819            sigPath[sigPathLen-1] =  '\0';



820        }



821       /* need to free for every iteration of the loop, but also have



822        * the free below EXIT_POINT in case of error */



823        utFree(blockPath);



824        blockPath = NULL;



825        utAssert(sigPath[sigPathLen-1] == '\0');



826        sigBlockName[*sigIdx] = sigPath; /* caller is responsible for free */



827



828        /* Label */



829        sigLabel[*sigIdx] = rtwCAPI_GetSignalName(signals, i);



830



831        /* Width */



832        mapIdx = rtwCAPI_GetSignalDimensionIdx(signals, i);



833        utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );



834        mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);



835        sigWidth[*sigIdx] = dimArray[mapIdx] * dimArray[mapIdx+1];



836



837        /* DataType and logDataType */



838        mapIdx = rtwCAPI_GetSignalDataTypeIdx(signals, i);



839        sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);



840        /* this mimics code in simulink.dll:mapSigDataTypeToLogDataType */



841        switch (sigDataType[*sigIdx]) {



842          case SS_DOUBLE:



843          case SS_SINGLE:



844          case SS_INT8:



845          case SS_UINT8:



846          case SS_INT16:



847          case SS_UINT16:



848          case SS_INT32:



849          case SS_UINT32:



850          case SS_BOOLEAN:



851            logDataType[*sigIdx] = sigDataType[*sigIdx];



852            break;



853          case SS_ENUM_TYPE:



854            logDataType[*sigIdx] = SS_INT32;



855            break;



856          default:



857            logDataType[*sigIdx] = SS_DOUBLE;



858            break;



859        }



860



861        /* Complexity */



862        sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);



863



864        /* Data Access - Pointer or Direct*/



865        isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));



866



867        /* Address */



868        mapIdx = rtwCAPI_GetSignalAddrIdx(signals, i);



869



870        rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,



871                                  sigDataAddr, sigIdx, mapIdx, dataAddrMap);



872



873        /* CrossingModelBoundary */



874        sigCrossMdlRef[*sigIdx] = crossingModel;



875



876        ++(*sigIdx);



877    }



878



879  EXIT_POINT:



880    utFree(blockPath);



881    return(errstr);



882



883} /* rtwCAPI_GetSigLogRecordInfo */

886/** Function: rtwCAPI_CountSysRan ==============================================



887 *   Recursive function that counts the number of non-NULL pointers in the array



888 *   of system ran dwork pointers, for the given MMI and below



889 */



890void rtwCAPI_CountSysRan(const rtwCAPI_ModelMappingInfo *mmi,



891			 int                            *count)



892{



893    sysRanDType **sysRan;



894    int numSys;



895    int nCMMI;



896    int numSysRan = 0;



897    int i;



898



899    if (mmi == NULL) return;



900



901    sysRan = rtwCAPI_GetSystemRan(mmi);



902    numSys    = rtwCAPI_GetNumSystems(mmi);



903    nCMMI     = rtwCAPI_GetChildMMIArrayLen(mmi);



904



905    /* Recurse over children */



906    for (i = 0; i < nCMMI; i++) {



907        rtwCAPI_CountSysRan(rtwCAPI_GetChildMMI(mmi,i), &numSysRan);



908    }



909



910    /* Count number of dworks in this MMI - skip root */



911    for (i = 1; i< numSys; i++) {



912        if (sysRan[i] != NULL) numSysRan++;



913    }



914



915    *count += numSysRan;



916



917} /* end rtwCAPI_CountSysRan */

920/** Function: rtwCAPI_FillSysRan ===============================================



921 *   Recursive function that fills in the system ran dwork pointers and their



922 *   corresponding tids in the given array, for the given MMI and below.  The



923 *   array to be filled in must be allocated outside



924 */



925void rtwCAPI_FillSysRan(const rtwCAPI_ModelMappingInfo *mmi,



926			sysRanDType                    **sysRan,



927			int                            *sysTid,



928                        int                            *fillIdx)



929{



930    int       numSys;



931    sysRanDType **mmiSysRan;



932    int       nCMMI;



933    int       *mmiSysTid;



934    int       idx         = *fillIdx;



935    const int *mmiConSys;



936    int i;



937



938    if (mmi == NULL) return;



939



940    numSys      = rtwCAPI_GetNumSystems(mmi);



941    mmiSysRan = rtwCAPI_GetSystemRan(mmi);



942    nCMMI       = rtwCAPI_GetChildMMIArrayLen(mmi);



943    mmiSysTid  = rtwCAPI_GetSystemTid(mmi);



944    mmiConSys  = rtwCAPI_GetContextSystems(mmi);



945



946    /* Recurse over children */



947    for (i = 0; i < nCMMI; i++) {



948        rtwCAPI_FillSysRan(rtwCAPI_GetChildMMI(mmi,i), sysRan, sysTid, &idx);



949    }



950



951    /* Populate arrays with dwork pointers and tid - skip root */



952    for (i = 1; i< numSys; i++) {



953        if (mmiSysRan[i] != NULL) {



954            idx++;



955            sysRan[idx] = mmiSysRan[i];



956            sysTid[idx] = mmiSysTid[mmiConSys[i]];



957        }



958    }



959



960    *fillIdx = idx;



961



962} /* end rtwCAPI_FillSysRan */
